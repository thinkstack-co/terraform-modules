
# Route53 Zone
Utilized to create a Route53 Domain Zone

# Usage
    module "route53_zone" {
      source  = "github.com/thinkstack-co/terraform-modules//modules/aws/route53/zone"
      
      comment = "ThinkStack primary domain"
      name    = "thinkstack.co"
      
      tags    = {
        terraform   = "yes"
        created_by  = "Zachary Hill"
        environment = "prod"
        role        = "external dns"
      }
    }
